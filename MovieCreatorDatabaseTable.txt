CREATE TABLE public."Movie"
(
    title varchar(30) NOT NULL,
    yearCreation varchar(14) NOT NULL,
    releaseDate varchar(14) NOT NULL,
    price decimal NOT NULL,
    nameStudio varchar(20) NOT NULL,
    nameDirector varchar(20) NOT NULL,
    description varchar(50) NOT NULL,
    nameMainActor varchar(20) NOT NULL,
    rented bit NOT NULL,
    PRIMARY KEY (title)
)
